    import MANIFEST from "./parser-reference.json";

import { tokenize } from "./tokenizer";

import { reportParseError } from "./error-reporting";

export function parse(subject) {
  const manifest = MANIFEST;
  const reference = manifest.reference;
  const dataTypes = manifest.dataType;
  const report = reportParseError;
  const rpn = [];

  let index = 0;
  let rpnLength = 0;
  let isExpectingOperand = true;
  let line = 1;
  let stack = null;
  let enclosure = [null, null];
  let enclosureNode = null;
  let hasOperand = false;
  let output;
  let stackPop;
  let stackPush;
  let pushEnclosure;
  let popEnclosure;
  let value;
  let rule;
  let from;
  let to;
  let token;
  let node;
  let definition;
  let type;
  let item;
  let precedence;
  let separator;
  let operands;

  /* eslint no-labels: 0 */
  loop: for (token = tokenize(subject, index); token; token = tokenize(subject, index)) {
    rule = token[0];
    value = token[1];
    definition = reference[rule];
    type = definition[0];
    from = index;
    index = to = token[2];
    line += token[3];
    token = rule;

    // preprocess
    switch (type) {
    case 0: continue loop;
    case 3:
      definition = isExpectingOperand
        ? definition[1]
        : definition[2];
      type = definition[0];
      break;
    }

    token = rule;
    rule = definition[1];

    // node
    node = {
      rule,
      token,
      value,
      parameterized: false,
      operands: 0,
      precedence: 0,
      augmented: true,
      from,
      to,
      line
    };

    pushEnclosure = popEnclosure = stackPop = stackPush = output = hasOperand = false;

    switch (type) {
    case 10: // operand
      node.augmented =
        isExpectingOperand = false;

      output = true;
      break;

    case 1: // unary
      node.precedence = precedence = definition[3];
      node.operands = 1;
      isExpectingOperand =
        stackPop =
        stackPush = true;
      break;

    case 2: // binary operator
      node.precedence = precedence = definition[3];
      node.operands = 2;
      isExpectingOperand =
        stackPop =
        stackPush = true;
      break;

    case 20:
      precedence = 0;
      stackPop = true;
    }

    // console.log("node: ", node.rule, isExpectingOperand);

    // pop until
    if (stackPop) {
      for (; stack; stack = stack[0]) {
        item = stack[1];
        if (!item.parameterized && item.precedence >= precedence) {
          if (item.rule) {
            rpn[rpnLength++] = item;
          }
          continue;
        }
        break;
      }
    }

    if (stackPush) {
      stack = [
        stack,
        node
      ];
    }
    else if (output && node.rule) {
      rpn[rpnLength++] = node;
    }
  }

  console.log("stack: ", JSON.stringify(stack, null, 2));

  console.log("rpn: ", rpn);

  return rpn;
}
