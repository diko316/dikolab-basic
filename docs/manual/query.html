    Simple extract value

    const object = require("data.json");

// returns 1
query(
  object,
  "get data[0].id"
);

    Compose Object

    // returns { "found": 1 }
query(
  object,
`get
  data[0].id as found,
  data[1].name as nextName
`
);

    List Filter

    // returns [ "en", "fr" ]
query(
  object,
  "get data[] #name=name1 language[]"
);

    Substitution

    // returns [ "en", "fr" ]
query(
  object,
  "get data[] #name=? language[]",
  [
    "name1"
  ]
);
